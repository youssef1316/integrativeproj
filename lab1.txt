void q4() {

    for (int i = 0; i < 4; i++) {

        sum += grades;

void q8() {
    cout << "Enter number: \n";
    int n; cin>>n;
    int is_prime[n];
    for (int i = 0; i < n; i++) {
        is_prime[i] = 1;
    }
    for (int i = 2; i < n; i++) {
        if (is_prime[i]) {
            for (int j = i * i; j < n; j += i) {
                is_prime[j] = 0;
            }
        }
    }
    for (int i = 2; i < n; i++) {
        if (is_prime[i]) {
            cout << i << "\n";
        }
    }
}
